function AfterReport return boolean is
CURSOR DO_CURSOR IS 
SELECT DOH.DO_HDR_ID FROM XXAKG_DEL_ORD_HDR DOH,XXAKG_DEL_ORD_DO_LINES DOL
		WHERE  DOH.DO_HDR_ID= DOL.DO_HDR_ID
		AND  DOH.ORG_ID= DOL.ORG_ID
		AND DOH.DO_STATUS ='GENERATED'
		AND DOH.ORG_ID=:P_ORG_ID
		AND DOL.WAREHOUSE_ORG_CODE ='SCI'
		AND READY_FOR_PRINTING ='Y'
		AND NVL(PRINTED_FLAG,'N')='N'
		AND DOL.ORG_ID=:P_ORG_ID
		AND DOH.DO_NUMBER BETWEEN :P_DO_FROM AND :P_DO_TO
		AND MODE_OF_TRANSPORT=NVL(:P_MODE_OF_TRANSPORT,MODE_OF_TRANSPORT)
		AND SUBSTR(ITEM_NUMBER,1,4)<>'SCRP'
GROUP BY DOH.DO_HDR_ID;
begin
 FOR I IN DO_CURSOR LOOP
  UPDATE XXAKG_DEL_ORD_HDR 
  SET PRINTED_FLAG='Y'
  WHERE DO_HDR_ID=I.DO_HDR_ID
AND ORG_ID=:P_ORG_ID;
END LOOP;
  return (TRUE);
 EXCEPTION
 	WHEN OTHERS THEN
 	RAISE;
end;
