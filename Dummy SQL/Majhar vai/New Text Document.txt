SELECT 
dbm.REGION_NAME,
dbm.CUSTOMER_NUMBER,
dbm.CUSTOMER_NAME,
doh.DO_HDR_ID,
doh.ORG_ID,
--doh.CUSTOMER_NUMBER,
doh.DO_NUMBER,
doh.DO_DATE,
--doh.CUSTOMER_NAME,
doh.MODE_OF_TRANSPORT,
doh.FINAL_DESTINATION,
mot.MOV_ORD_HDR_ID,
mot.DO_NUMBER,
moh.MOV_ORD_HDR_ID,
moh.MOV_ORDER_NO,
moh.WAREHOUSE_ORG_CODE
FROM 
apps.XXAKG_DISTRIBUTOR_BLOCK_M dbm,
apps.XXAKG_DEL_ORD_HDR doh,
apps.XXAKG_MOV_ORD_HDR moh,
apps.XXAKG_MOV_ORD_DTL mot
where rownum=1
and doh.CUSTOMER_NUMBER=doh.CUSTOMER_NUMBER
and moh.MOV_ORD_HDR_ID=mot.MOV_ORD_HDR_ID;
