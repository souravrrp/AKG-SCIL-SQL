SELECT * FROM xxakgmain.XXAKG_CER_ITEM_CONVERSION xcic
WHERE NOT EXISTS (
select 1
from apps.MTL_UOM_CLASS_CONVERSIONS muc,  apps.mtl_system_items_b msi
where msi.INVENTORY_ITEM_ID=muc.INVENTORY_ITEM_ID and msi.organization_id=99 AND XCIC.ITEM_ID=muc.INVENTORY_ITEM_ID
AND XCIC.FROM_UOM_CODE=MUC.FROM_UOM_CODE AND XCIC.TO_UOM_CODE=MUC.TO_UOM_CODE AND XCIC.CONVERSION_RATE=MUC.CONVERSION_RATE
)--621
and not exists (
select 1 from apps.MTL_UOM_CLASS_CONVERSIONS muc1 where muc1.INVENTORY_ITEM_ID=xcic.ITEM_ID)
