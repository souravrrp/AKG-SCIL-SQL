SELECT  OAP.ORGANIZATION_ID, OOD.ORGANIZATION_CODE, OOD.ORGANIZATION_NAME, OOD.OPERATING_UNIT, OAP.STATUS, OAP.PERIOD_NAME, OAP.PERIOD_NUMBER, OAP.CLOSE_DATE
FROM APPS.ORG_ACCT_PERIODS_V OAP, APPS.ORG_ORGANIZATION_DEFINITIONS OOD
WHERE OAP.ORGANIZATION_ID=OOD.ORGANIZATION_ID
AND PERIOD_NAME = 'MAR-15'
ORDER BY 1
