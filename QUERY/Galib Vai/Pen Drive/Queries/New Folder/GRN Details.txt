    RVT.ORGANIZATION_ID, 
    RVT.CREATION_DATE, 
    RVT.CREATED_BY, 
    RVT.PARENT_TRANSACTION_ID, 
    RVT.SHIPMENT_HEADER_ID, 
    RVT.TRANSACTION_ID, 
    RVT.TRANSACTION_TYPE, 
    RVT.TRANSACTION_DATE, 
    RVT.TRANSACT_QTY, 
    RVT.TRANSACT_UOM, 
    RVT.ITEM_DESC, 
    RVT.DESTINATION_TYPE_CODE, 
    RVT.PO_HEADER_ID, 
    RVT.PO_LINE_ID, 
    RVT.PO_LINE_LOCATION_ID, 
    RVT.SHIP_TO_LOCATION,
    RVT.EMPLOYEE_ID, 
    RVT.RECEIVER, 
    RVT.DELIVER_TO_PERSON, 
    RVT.SUPPLIER_ID, 
    RVT.SUPPLIER, 
    RVT.VENDOR_NAME, 
    RVT.RECEIPT_NUM, 
    RVT.RECEIPT_DATE,
    POL.QUANTITY,
    POL.UNIT_PRICE,
    POL.UNIT_PRICE*POL.QUANTITY TOTAL_VALUES
FROM 
    APPS.RCV_VRC_TXS_V RVT,
    APPS.PO_LINES_ALL POL
--    APPS.PO_HEADERS_ALL PH
WHERE 
    RVT.PO_LINE_ID = POL.PO_LINE_ID
    AND SUPPLIER_ID = 48009
--RECEIPT_NUM = 1169
--WHERE TO_ORGANIZATION_ID = 101
AND trunc(RVT.CREATION_DATE) BETWEEN '01-JAN-2011' AND '01-JUN-2015'
GROUP BY
    RVT.ORGANIZATION_ID, 
    RVT.CREATION_DATE, 
    RVT.CREATED_BY, 
    RVT.PARENT_TRANSACTION_ID, 
    RVT.SHIPMENT_HEADER_ID, 
    RVT.TRANSACTION_ID, 
    RVT.TRANSACTION_TYPE, 
    RVT.TRANSACTION_DATE, 
    RVT.TRANSACT_QTY, 
    RVT.TRANSACT_UOM, 
    RVT.ITEM_DESC, 
    RVT.DESTINATION_TYPE_CODE, 
    RVT.PO_HEADER_ID, 
    RVT.PO_LINE_ID, 
    RVT.PO_LINE_LOCATION_ID, 
    RVT.SHIP_TO_LOCATION,
    RVT.EMPLOYEE_ID, 
    RVT.RECEIVER, 
    RVT.DELIVER_TO_PERSON, 
    RVT.SUPPLIER_ID, 
    RVT.SUPPLIER, 
    RVT.VENDOR_NAME, 
    RVT.RECEIPT_NUM, 
    RVT.RECEIPT_DATE,
    POL.QUANTITY,
    POL.UNIT_PRICE,
    POL.UNIT_PRICE*POL.QUANTITY
