SELECT 
    MTL.REQUEST_NUMBER, 
    MTL.QUANTITY,
    MTL.UOM_CODE,
    MTL.TRANSACTION_TYPE_NAME,
    MTL.MOVE_ORDER_TYPE_NAME,
    MTH.HEADER_STATUS_NAME,
    MTL.ORGANIZATION_ID,
    MTL.FROM_SUBINVENTORY_CODE,
    MTL.STATUS_DATE,
    MTL.CREATION_DATE,
    MTH.ATTRIBUTE3,
    MTH.ATTRIBUTE4,
    MTH.ATTRIBUTE_CATEGORY   
FROM 
    APPS.MTL_TXN_REQUEST_LINES_V MTL,
    APPS.MTL_TXN_REQUEST_HEADERS_V MTH
WHERE
    MTL.ORGANIZATION_ID = MTH.ORGANIZATION_ID
    AND MTL.HEADER_ID = MTH.HEADER_ID
    AND MTL.ORGANIZATION_ID = 89
    AND MTL.FROM_SUBINVENTORY_CODE = 'AKC-IT'
    AND MTL.TRANSACTION_TYPE_NAME = 'Move Order Issue'
    AND TRUNC (MTL.CREATION_DATE) BETWEEN '01-AUG-2015' AND '30-SEP-2015'
--and rownum<10
