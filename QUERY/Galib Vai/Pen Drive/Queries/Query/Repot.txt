select  a.DO_NUMBER,
                    a.CONFIRM_DATE,
                    b.MOV_ORDER_NO,
                    b.CONFIRMED_DATE,
                    b.TRANSPORT_MODE,
                    c.REGION_NAME
from
XXAKG.XXAKG_MOV_ORD_DTL a,
XXAKG.XXAKG_MOV_ORD_HDR b,
APPS.XXAKG_DISTRIBUTOR_BLOCK_M c
where  a.MOV_ORD_HDR_ID=b.MOV_ORD_HDR_ID
and c.CUSTOMER_NUMBER=a.CUSTOMER_NUMBER
and DO_NUMBER='DO/SCOU/291002'
and rownum<5;

select hl.ORG_ID,
hl.ORDER_NUMBER,
la.ORDERED_ITEM,
la.ORDERED_QUANTITY,
dh.DO_STATUS,
dh.DO_NUMBER,
dh.CUSTOMER_NUMBER,
dh.CUSTOMER_NAME,
dl.SHIP_TO_LOCATION,
dl.ACTUAL_RETAILER_SHIP_TO,
mh.MOV_ORDER_NO,
mh.FINAL_DESTINATION,
mh.CONFIRMED_DATE
from 
APPS.oe_order_headers_all hl,
APPS.OE_ORDER_LINES_ALL la,
APPS.XXAKG_DEL_ORD_HDR dh,
APPS.XXAKG_DEL_ORD_DO_LINES dl,
XXAKG.XXAKG_MOV_ORD_DTL md,
XXAKG.XXAKG_MOV_ORD_HDR mh
where
hl.HEADER_ID=la.HEADER_ID
and la.SHIPMENT_PRIORITY_CODE=dl.DO_NUMBER
and dl.DO_HDR_ID=dh.DO_HDR_ID
and dh.DO_NUMBER=dl.DO_NUMBER
and md.MOV_ORD_HDR_ID=mh.MOV_ORD_HDR_ID
and md.DO_NUMBER=dh.DO_NUMBER
and MOV_ORDER_NO='MO/SCOU/000003'
and rownum<35;
